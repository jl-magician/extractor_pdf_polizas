    ```python
    # Evaluation stats (Phase 16 — D-15b)
    total_evaluated = db.execute(
        select(func.count(Poliza.id)).where(Poliza.evaluation_score.isnot(None))
    ).scalar() or 0

    avg_score = db.execute(
        select(func.avg(Poliza.evaluation_score)).where(Poliza.evaluation_score.isnot(None))
    ).scalar()

    eval_pct = round((total_evaluated / total_polizas * 100), 1) if total_polizas > 0 else 0.0
    avg_score_display = round(avg_score * 100, 1) if avg_score is not None else None
    ```
