```python
@patch("policy_extractor.api.upload.SessionLocal")
@patch("policy_extractor.api.upload.orm_to_schema")
@patch("policy_extractor.api.upload.upsert_policy")
@patch("policy_extractor.api.upload.extract_policy")
@patch("policy_extractor.api.upload.ingest_pdf")
@patch("policy_extractor.api.upload.is_already_extracted", return_value=False)
@patch("policy_extractor.api.upload.compute_file_hash", return_value="abc123")
def test_pipeline_success_sets_complete(mock_hash, mock_already, mock_ingest, mock_extract, mock_upsert, mock_orm, mock_session, tmp_path):
    from policy_extractor.api import upload
    # Setup
    pdf_file = tmp_path / "test.pdf"
    pdf_file.write_bytes(b"%PDF-1.4 test")
    job = upload._create_job("test.pdf")

    # Mock extract_policy returning a fake PolicyExtraction
    fake_extraction = MagicMock()
    fake_extraction.model_dump.return_value = {"numero_poliza": "POL-001"}
    mock_extract.return_value = (fake_extraction, MagicMock())
    mock_session.return_value = MagicMock()

    # Run extraction directly (not in thread)
    upload._run_extraction(job["job_id"], pdf_file, None, False)

    # Verify
    result_job = upload._get_job(job["job_id"])
    assert result_job["status"] == "complete"
    assert result_job["result"] is not None
    mock_ingest.assert_called_once()
    mock_extract.assert_called_once()
    mock_upsert.assert_called_once()
```

Note: The mock patches target the IMPORT locations inside `upload.py`. Since `_run_extraction` uses lazy imports (imports inside the function body), the patch targets must match the module where they're imported. The actual implementation in Plan 01 uses lazy imports like `from policy_extractor.ingestion import ingest_pdf` inside `_run_extraction`. Patch at the source: `policy_extractor.ingestion.ingest_pdf`, `policy_extractor.extraction.extract_policy`, etc. Check the actual import paths in the `_run_extraction` function and patch accordingly.
